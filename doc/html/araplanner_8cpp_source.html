00249 void ARAPlanner::UpdatePreds(ARAState* state, ARASearchStateSpace_t* pSearchStateSpace)
00250 {
00251     vector<int> PredIDV;
00252     vector<int> CostV;
00253     CKey key;
00254     ARAState *predstate;
00255 
00256     environment_->GetPreds(state->MDPstate->StateID, &PredIDV, &CostV);
00257 
00258     //iterate through predecessors of s
00259     for (int pind = 0; pind < (int)PredIDV.size(); pind++) {
00260         CMDPSTATE* PredMDPState = GetState(PredIDV[pind], pSearchStateSpace);
00261         predstate = (ARAState*)(PredMDPState->PlannerSpecificData);
00262         if (predstate->callnumberaccessed != pSearchStateSpace->callnumber) {
00263             ReInitializeSearchStateInfo(predstate, pSearchStateSpace);
00264         }
00265 
00266         //see if we can improve the value of predstate
00267         if (predstate->g > state->v + CostV[pind]) {
00268             predstate->g = state->v + CostV[pind];
00269             predstate->bestnextstate = state->MDPstate;
00270             predstate->costtobestnextstate = CostV[pind];
00271 
00272             //re-insert into heap if not closed yet
00273             if (predstate->iterationclosed != pSearchStateSpace->searchiteration) {
00274                 key.key[0] = predstate->g + (int)(pSearchStateSpace->eps * predstate->h);
00275                 //key.key[1] = predstate->h;
00276                 if (predstate->heapindex != 0)
00277                     pSearchStateSpace->heap->updateheap(predstate, key);
00278                 else
00279                     pSearchStateSpace->heap->insertheap(predstate, key);
00280             }
00281             //take care of incons list
00282             else if (predstate->listelem[ARA_INCONS_LIST_ID] == NULL) {
00283                 pSearchStateSpace->inconslist->insert(predstate, ARA_INCONS_LIST_ID);
00284             }
00285         }
00286     } //for predecessors
00287 }

00349     CKey goalkey;
00350 
00351     expands = 0;

00378

00890 bool ARAPlanner::Search(ARASearchStateSpace_t* pSearchStateSpace, vector<int>& pathIds, int & PathCost,
00891                         bool bFirstSolution, bool bOptimalSolution, double MaxNumofSecs)
00892 {
00893     CKey key;
00894     TimeStarted = clock();
00895     searchexpands = 0;
00896     double old_repair_time = repair_time;
00897     if (!use_repair_time) repair_time = MaxNumofSecs;
00898 
00899 #if DEBUG
00900     SBPL_FPRINTF(fDeb, "new search call (call number=%d)\n", pSearchStateSpace->callnumber);
00901 #endif
00902 
00903     if (pSearchStateSpace->bReinitializeSearchStateSpace == true) {
00904         //re-initialize state space
00905         ReInitializeSearchStateSpace(pSearchStateSpace);
00906     }
00907 
00908     if (bOptimalSolution) {
00909         pSearchStateSpace->eps = 1;
00910         MaxNumofSecs = INFINITECOST;
00911         repair_time = INFINITECOST;
00912     }
00913     else if (bFirstSolution) {
00914         MaxNumofSecs = INFINITECOST;
00915         repair_time = INFINITECOST;
00916     }
00917 
00918     //ensure heuristics are up-to-date
00919     environment_->EnsureHeuristicsUpdated((bforwardsearch == true));
00920 
00921     //the main loop of ARA*
00922     stats.clear();
00923     int prevexpands = 0;
00924     clock_t loop_time;
00925     while (pSearchStateSpace->eps_satisfied > final_epsilon &&
00926            (clock() - TimeStarted) < MaxNumofSecs * (double)CLOCKS_PER_SEC &&
00927                (pSearchStateSpace->eps_satisfied == INFINITECOST ||
00928                (clock() - TimeStarted) < repair_time * (double)CLOCKS_PER_SEC))
00929     {
00930         loop_time = clock();
00931         //decrease eps for all subsequent iterations
00932         if (fabs(pSearchStateSpace->eps_satisfied - pSearchStateSpace->eps) < ERR_EPS && !bFirstSolution) {
00933             pSearchStateSpace->eps = pSearchStateSpace->eps - dec_eps;
00934             if (pSearchStateSpace->eps < final_epsilon) pSearchStateSpace->eps = final_epsilon;
00935 
00936             //the priorities need to be updated
00937             pSearchStateSpace->bReevaluatefvals = true;
00938 
00939             //it will be a new search
00940             pSearchStateSpace->bNewSearchIteration = true;
00941 
00942             //build a new open list by merging it with incons one
00943             BuildNewOPENList(pSearchStateSpace);
00944         }
00945 
00946         if (pSearchStateSpace->bNewSearchIteration) {
00947             pSearchStateSpace->searchiteration++;
00948             pSearchStateSpace->bNewSearchIteration = false;
00949         }
00950 
00951         //re-compute f-values if necessary and reorder the heap
00952         if (pSearchStateSpace->bReevaluatefvals) Reevaluatefvals(pSearchStateSpace);
00953 
00954         //improve or compute path
00955         if (ImprovePath(pSearchStateSpace, MaxNumofSecs) == 1) {
00956             pSearchStateSpace->eps_satisfied = pSearchStateSpace->eps;
00957         }
00958 
00959         //print the solution cost and eps bound
00960         SBPL_PRINTF("eps=%f expands=%d g(searchgoal)=%d time=%.3f\n", pSearchStateSpace->eps_satisfied,
00961                     searchexpands - prevexpands,
00962                     ((ARAState*)pSearchStateSpace->searchgoalstate->PlannerSpecificData)->g,
00963                     double(clock() - loop_time) / CLOCKS_PER_SEC);
00964 
00965         if (pSearchStateSpace->eps_satisfied == finitial_eps && pSearchStateSpace->eps == finitial_eps) {
00966             finitial_eps_planning_time = double(clock() - loop_time) / CLOCKS_PER_SEC;
00967             num_of_expands_initial_solution = searchexpands - prevexpands;
00968         }
00969 
00970         if (stats.empty() || pSearchStateSpace->eps_satisfied != stats.back().eps) {
00971             PlannerStats tempStat;
00972             tempStat.eps = pSearchStateSpace->eps_satisfied;
00973             tempStat.expands = searchexpands - prevexpands;
00974             tempStat.time = double(clock() - loop_time) / CLOCKS_PER_SEC;
00975             tempStat.cost = ((ARAState*)pSearchStateSpace->searchgoalstate->PlannerSpecificData)->g;
00976             stats.push_back(tempStat);
00977         }
00978 
00979 #if DEBUG
00980         SBPL_FPRINTF(fDeb, "eps=%f expands=%d g(searchgoal)=%d time=%.3f\n", pSearchStateSpace->eps_satisfied,
00981                      searchexpands - prevexpands,
00982                      ((ARAState*)pSearchStateSpace->searchgoalstate->PlannerSpecificData)->g,
00983                      double(clock()-loop_time)/CLOCKS_PER_SEC);
00984         PrintSearchState((ARAState*)pSearchStateSpace->searchgoalstate->PlannerSpecificData, fDeb);
00985 #endif
00986         prevexpands = searchexpands;
00987 
00988         //if just the first solution then we are done
00989         if (bFirstSolution) break;
00990 
00991         //no solution exists
00992         if (((ARAState*)pSearchStateSpace->searchgoalstate->PlannerSpecificData)->g == INFINITECOST) break;
00993     }
00994     repair_time = old_repair_time;
00995 
00996 #if DEBUG
00997     SBPL_FFLUSH(fDeb);
00998 #endif
00999 
01000     PathCost = ((ARAState*)pSearchStateSpace->searchgoalstate->PlannerSpecificData)->g;
01001     MaxMemoryCounter += environment_->StateID2IndexMapping.size() * sizeof(int);
01002 
01003     SBPL_PRINTF("MaxMemoryCounter = %d\n", MaxMemoryCounter);
01004 
01005     int solcost = INFINITECOST;
01006     bool ret = false;
01007     if (PathCost == INFINITECOST) {
01008         SBPL_PRINTF("could not find a solution\n");
01009         ret = false;
01010     }
01011     else {
01012         SBPL_PRINTF("solution is found\n");
01013         pathIds = GetSearchPath(pSearchStateSpace, solcost);

01018                 searchexpands, (clock() - TimeStarted) / ((double)CLOCKS_PER_SEC), solcost);
