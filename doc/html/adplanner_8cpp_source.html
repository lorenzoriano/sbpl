00255 void ADPlanner::UpdateSetMembership(ADState* state)
00256 {
00257     CKey key;
00258 
00259     if (state->v != state->g) {
00260         if (state->iterationclosed != pSearchStateSpace_->searchiteration) {
00261             key = ComputeKey(state);
00262             if (state->heapindex == 0) {
00263                 //need to remove it because it can happen when updating edge costs and state is in incons
00264                 if (state->listelem[AD_INCONS_LIST_ID] != NULL) {
00265                     pSearchStateSpace_->inconslist->remove(state, AD_INCONS_LIST_ID);
00266                 }
00267 
00268                 pSearchStateSpace_->heap->insertheap(state, key);
00269 
00270             }
00271             else
00272                 pSearchStateSpace_->heap->updateheap(state, key);
00273         }
00274         else if (state->listelem[AD_INCONS_LIST_ID] == NULL) {
00275             pSearchStateSpace_->inconslist->insert(state, AD_INCONS_LIST_ID);
00276         }
00277     }
00278     else {
00279         if (state->heapindex != 0)
00280             pSearchStateSpace_->heap->deleteheap(state);
00281         else if (state->listelem[AD_INCONS_LIST_ID] != NULL) {
00282             pSearchStateSpace_->inconslist->remove(state, AD_INCONS_LIST_ID);
00283         }
00284     }

00287 void ADPlanner::Recomputegval(ADState* state)
00288 {
00289     vector<int> searchpredsIDV; //these are predecessors if search is done forward and successors otherwise
00290     vector<int> costV;
00291     CKey key;
00292     ADState *searchpredstate;
00293 
00294     if (bforwardsearch)
00295         environment_->GetPreds(state->MDPstate->StateID, &searchpredsIDV, &costV);
00296     else
00297         environment_->GetSuccs(state->MDPstate->StateID, &searchpredsIDV, &costV);
00298 
00299     //iterate through predecessors of s and pick the best
00300     state->g = INFINITECOST;
00301     for (int pind = 0; pind < (int)searchpredsIDV.size(); pind++) {
00302         //skip the states that do not exist - they can not be used to improve g-value anyway
00303         if (environment_->StateID2IndexMapping[searchpredsIDV[pind]][ADMDP_STATEID2IND] == -1) continue; 
00304 
00305         CMDPSTATE* predMDPState = GetState(searchpredsIDV[pind], pSearchStateSpace_);
00306         int cost = costV[pind];
00307         searchpredstate = (ADState*)(predMDPState->PlannerSpecificData);
00308 
00309         //see if it can be used to improve
00310         if (searchpredstate->callnumberaccessed == pSearchStateSpace_->callnumber && state->g > searchpredstate->v
00311             + cost) {
00312             if (bforwardsearch) {
00313                 state->g = searchpredstate->v + cost;
00314                 state->bestpredstate = predMDPState;
00315             }
00316             else {
00317                 state->g = searchpredstate->v + cost;
00318                 state->bestnextstate = predMDPState;
00319                 state->costtobestnextstate = cost;
00320             }
00321         }
00322     } //over preds
00323 }
