00547 void EnvironmentNAV2DUU::GetPreds(int stateID, const vector<sbpl_BinaryHiddenVar_t>* updatedhvaluesV,
00548                                   vector<CMDPACTION>* IncomingDetActionV, vector<CMDPACTION>* IncomingStochActionV,
00549                                   vector<sbpl_BinaryHiddenVar_t>* StochActionNonpreferredOutcomeV)
00550 {
00551     int aind;
00552 
00553     //get state coords
00554     int destx = ENVNAV2DUU_STATEIDTOX(stateID);
00555     int desty = ENVNAV2DUU_STATEIDTOY(stateID);
00556 
00557     //clear succs
00558     IncomingDetActionV->clear();
00559     IncomingStochActionV->clear();
00560     StochActionNonpreferredOutcomeV->clear();
00561 
00562     //check that the destination is valid
00563     if (!IsWithinMapCell(destx, desty) || EnvNAV2DUUCfg.Grid2D[destx][desty] >= EnvNAV2DUUCfg.obsthresh) return;
00564 
00565     //see if the destination was originally uncertain
00566     bool bDet = false;
00567     //no need to worry about ==1.0 since obstacles are rejected above
00568     if (EnvNAV2DUUCfg.UncertaintyGrid2D[destx][desty] < NAV2DUU_ERR_EPS) 
00569     bDet = true;
00570 
00571     //if yes, then figure out the probability of being an obstacle
00572     float ProbObs = EnvNAV2DUUCfg.UncertaintyGrid2D[destx][desty];
00573     int desth_ID = EnvNAV2DUUCfg.HiddenVariableXY2ID[destx][desty];
00574     bool bFreeh = false;
00575 
00576     //iterate over updated h-values to make sure dest cell is not in there
00577     for (int hind = 0; hind < (int)updatedhvaluesV->size(); hind++) {
00578         if (updatedhvaluesV->at(hind).Prob < NAV2DUU_ERR_EPS) bFreeh = true; //there are elements that are free
00579 
00580         if (updatedhvaluesV->at(hind).h_ID == desth_ID) {
00581             ProbObs = updatedhvaluesV->at(hind).Prob; //found
00582         }
00583     }
00584 
00585     //if now known to be an obstacle, then no preds
00586     if (ProbObs > 1.0 - NAV2DUU_ERR_EPS)
00587         return;
00588     else if (ProbObs < NAV2DUU_ERR_EPS) bDet = false; //now it is a deterministic cell
00589 
00590     //get the destination costmult
00591     int destcostmult = EnvNAV2DUUCfg.Grid2D[destx][desty];
00592 
00593 #if DEBUG
00594     if (EnvNAV2DUUCfg.numofdirs > 8) {
00595         SBPL_ERROR("ERROR: number of directions can not exceed 8 for now\n");
00596         throw new SBPL_Exception();
00597     }
00598 #endif
00599 
00600     //iterate over neighbors
00601     for (aind = 0; aind < EnvNAV2DUUCfg.numofdirs; aind++) {
00602         //the actions are undirected, so we can use the same array of actions as in getsuccs case
00603         int predX = destx + EnvNAV2DUUCfg.dx_[aind];
00604         int predY = desty + EnvNAV2DUUCfg.dy_[aind];
00605 
00606         //skip the invalid cells
00607         if (!IsWithinMapCell(predX, predY) || EnvNAV2DUUCfg.Grid2D[predX][predY] >= EnvNAV2DUUCfg.obsthresh) continue;
00608 
00609         //running costmult
00610         int costmult = destcostmult;
00611 
00612         //get the costmultiplier of pred and update total costmult
00613         costmult = __max(costmult, EnvNAV2DUUCfg.Grid2D[predX][predY]);
00614 
00615         //if diagonal move
00616         if (predX != destx && predY != desty) {
00617             //check intersecting cells to make sure they are not uncertain
00618             if (EnvNAV2DUUCfg.UncertaintyGrid2D[destx][predY] >= NAV2DUU_ERR_EPS) {
00619                 if (!bFreeh) continue;
00620 
00621                 //check that it is not known to be free. Otherwise - invalid move
00622                 int tempID = EnvNAV2DUUCfg.HiddenVariableXY2ID[destx][predY];
00623                 bool btempfree = false;
00624                 for (int hind = 0; hind < (int)updatedhvaluesV->size(); hind++) {
00625                     if (updatedhvaluesV->at(hind).h_ID == tempID) {
00626                         if (updatedhvaluesV->at(hind).Prob < NAV2DUU_ERR_EPS) {
00627                             btempfree = true;
00628                             break;
00629                         }
00630                     }
00631                 }
00632                 if (!btempfree) continue;
00633             }
00634 
00635             if (EnvNAV2DUUCfg.UncertaintyGrid2D[predX][desty] >= NAV2DUU_ERR_EPS) {
00636                 if (!bFreeh) continue;
00637 
00638                 //check that it is not known to be free. Otherwise - invalid move
00639                 int tempID = EnvNAV2DUUCfg.HiddenVariableXY2ID[predX][desty];
00640                 bool btempfree = false;
00641                 for (int hind = 0; hind < (int)updatedhvaluesV->size(); hind++) {
00642                     if (updatedhvaluesV->at(hind).h_ID == tempID) {
00643                         if (updatedhvaluesV->at(hind).Prob < NAV2DUU_ERR_EPS) {
00644                             btempfree = true;
00645                             break;
00646                         }
00647                     }
00648                 }
00649                 if (!btempfree) continue;
00650             }
00651 
00652             //compute the costmultiplier of intersect cells and update total costmult
00653             costmult = __max(costmult, EnvNAV2DUUCfg.Grid2D[destx][predY]);
00654             costmult = __max(costmult, EnvNAV2DUUCfg.Grid2D[predX][desty]);
00655         }
00656 
00657         //make sure cost is still valid
00658         if (costmult >= EnvNAV2DUUCfg.obsthresh) continue;
00659 
00660         //otherwise compute the actual cost (once again we use the fact that
00661         //actions are undirected to determine the cost)
00662         int cost = (((int)costmult) + 1) * EnvNAV2DUUCfg.dxy_distance_mm_[aind];
00663 
00664         //create action
00665         int predstateID = ENVNAV2DUU_XYTOSTATEID(predX, predY);
00666         CMDPACTION action(aind, predstateID); //TODO-use reverseindex
00667 
00668         if (bDet) {
00669             //if dest is known - then form a deterministic action
00670             action.AddOutcome(stateID, cost, 1.0);
00671             IncomingDetActionV->push_back(action);
00672         }
00673         else {
00674             //if dest is unknown - then form a stoch action and compute the corresponding belief part
00675             action.AddOutcome(stateID, cost, 1 - ProbObs); //preferred outcome
00676             action.AddOutcome(predstateID, 2 * cost, ProbObs); //non-preferred outcome (stateID is untraversable)
00677             IncomingStochActionV->push_back(action);
00678             //also insert the corresponding hidden variable value
00679             sbpl_BinaryHiddenVar_t hval;
00680             hval.h_ID = desth_ID;
00681             hval.Prob = 1; //known to be an obstacle
00682             StochActionNonpreferredOutcomeV->push_back(hval);
00683         }
00684     }
00685 }
