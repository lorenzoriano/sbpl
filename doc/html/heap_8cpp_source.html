00073 {
00074     int child;
00075 
00076     if (currentsize != 0) {
00077         for (; 2 * hole <= currentsize; hole = child) {
00078             child = 2 * hole;

00080             if (child != currentsize && heap[child + 1].key < heap[child].key) ++child;
00081             if (heap[child].key < tmp.key) {
00082                 percolates += 1;
