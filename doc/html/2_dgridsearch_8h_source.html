00133     inline int getlowerboundoncostfromstart_inmm(int x, int y)
00134     {
00135         if (term_condition_usedlast == SBPL_2DGRIDSEARCH_TERM_CONDITION_OPTPATHFOUND) {
00136             //heuristic search
00137             int h = SBPL_2DGRIDSEARCH_HEUR2D(x,y);
00138             //the logic is that if s wasn't expanded, then g(s) + h(s) >=
00139             //maxcomputed_fval => g(s) >= maxcomputed_fval - h(s)
00140             return ((searchStates2D_[x][y].iterationaccessed == iteration_ &&
00141                     searchStates2D_[x][y].g + h <= largestcomputedoptf_)      ? searchStates2D_[x][y].g :
00142                     largestcomputedoptf_ < INFINITECOST                       ? largestcomputedoptf_ - h :
00143                                                                                 INFINITECOST);
00144         }
00145         else {
00146             //Dijkstra's search
00147             //the logic is that if s wasn't expanded, then g(s) >= maxcomputed_fval => g(s) >= maxcomputed_fval - h(s)
00148             return ((searchStates2D_[x][y].iterationaccessed == iteration_ &&
00149                     searchStates2D_[x][y].g <= largestcomputedoptf_) ? searchStates2D_[x][y].g :
00150                     largestcomputedoptf_);
00151         }
00152     }
